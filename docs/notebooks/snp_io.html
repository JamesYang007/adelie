[3]:

n = 4000
ps = [10000, 20000, 30000]
seed = 0

filenames = [
    f"/tmp/dummy_{i}.snpdat"
    for i in range(len(ps))
]

np.random.seed(seed)
calldatas = [
    ad.data.create_snp_unphased(n, pi, seed=i)["X"]
    for i, pi in enumerate(ps)
]
calldata = np.concatenate(calldatas, axis=-1)

for name, dat in zip(filenames, calldatas):
    handler = ad.io.snp_unphased(name)
    handler.write(dat, n_threads=4)
