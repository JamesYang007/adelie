X = ad.matrix.snp_unphased(filenames, n_threads=n_threads)
p = int(np.sum(ps))

np.random.seed(314)
beta = np.random.normal(0, 1, 3)
Xb = np.empty(n)
X.btmul(p-3, 3, beta, np.ones(n), Xb)
y = Xb + np.random.normal(0, 1, n)

[16]:

state = ad.grpnet(
    X=X,
    y=y,
    groups=16 * np.arange(p // 16, dtype=int),
    tol=1e-8,
    n_threads=n_threads,
    check_state=True,
)
